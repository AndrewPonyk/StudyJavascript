2)To understand difference between delegate()/undelegate() and live()/die()(on()/off()) see example : 
	           	// Using .live()

					$("table").each(function(){
					
					  $("td", this).live("hover", function(){
					
					    $(this).toggleClass("hover");
					
					  });
					
					});
					
					// Using .die()
					
					$("table").each(function(){
					  $("td", this).die("hover");
					});	
					=========================================	
					// Using .delegate()
					
					$("table").delegate("td", "hover", function(){
					
					  $(this).toggleClass("hover");
					
					});
					
					// Using .undelegate()
					$("table").undelegate("td", "hover");
